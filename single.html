Create a file named  doc_js.js 

 

playing_CK = 'playing_K';

const click_for_play_Sound = e => {
   
   const keyboard_key = e.keyCode;
   const key_n_Element = document.querySelector(`button[data-key="${keyboard_key}"]`);
   key_n_Element.classList.add(playing_CK);

   alert(keyboard_key);
};

function sounds_piano_keys(key)
{
  var key_p = 'sounds_piano_keys/'+ key + ".mp3";
  var audio = new Audio(key_p);
  audio.play();
}

const Return_to_normal = e => {
  e.target.classList.remove(playing_CK);
}

const e_buttons = Array.from(document.querySelectorAll('button'));
e_buttons.forEach(R_C);

function R_C(item){
  item.addEventListener('transitionend',Return_to_normal,false);
}

window.addEventListener('keydown' , click_for_play_Sound);

These codes return the piano key button that was pressed to its normal size
This is done by this function
const Return_to_normal = e => {
   e.target.classList.remove(playing_CK);
}
Calling this function deletes the class from the current piano key item so that the key button returns to its normal size
As shown in the picture

But you might want to know how this function is called and how the piano key item was returned to its normal size when you finished clicking

This is done by the transitionend event
where when the size of the key button reaches the size specified in the transform property

 transform:scale(1.20);

The Return_to_normal  function will be called based on the transitionend event
So the class is deleted from the piano key element and it will be returned to the normal size
